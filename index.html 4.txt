<script>
let problems = [
  "Print Hello World",
  "Add two numbers",
  "Check even or odd"
];

function loadProblem() {
  let index = document.getElementById("problemSelect").value;
  document.getElementById("problemText").innerText = problems[index];
}

function runCode() {
  let outputBox = document.getElementById("output");
  outputBox.innerHTML = "";

  let code = document.getElementById("code").value;

  try {
    let oldLog = console.log;
    let logs = [];

    console.log = function(msg) {
      logs.push(msg);
    };

    eval(code);

    console.log = oldLog;

    outputBox.innerHTML = logs.join("\n");
  } catch (err) {
    outputBox.innerHTML = "Error: " + err;
  }
}

loadProblem();
</script>
